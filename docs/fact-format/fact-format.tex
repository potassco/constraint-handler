\section{Syntax}

\subsection{Basic terms}
\begin{grammar}

<type> ::=
~\alt `bool' | `float' | `int' | `none' | `string' | `symbol'
\alt `function' | `set' | `multimap'

<bool> ::= `true' | `false'

<int> ::= any integer literal

<str> ::= any string literal (enclosed in quotes)

<name> ::= any suitable string

<term> ::= <int> | <str> | <symbol>

<terms> ::= <term> | <term> `,' <terms>

<symbol> ::= <name> | <name> `(' <terms> `)'

<term-list> ::= `(' `)' | `(' <term> `,' <term-list> `)'

%<constant> ::= `bool' `(' <bool> `)' \alt `int' `(' <int> `)' \alt `float' `(' <str> `)' \alt `string `(' <str> `)' \alt `symbol `(' <term> `)'

\end{grammar}

\subsection{Building expression terms}
\begin{grammar}

<operator> ::=
~\alt `python' `(' <str> `)'
\alt <lambda-expr>
\alt <variable>
\alt <bool-operator>
\alt <conditional-operator>
\alt <float-operator>
\alt <int-operator>
\alt <multimap-operator>
\alt <set-operator>
\alt <string-operator>

<eq-operator> ::= `eq' | `neq'

<comp-operator> ::= <eq-operator> | `leq' | `lt' | `geq' | `gt'

<bool-operator> ::=
~\alt <eq-operator> | `conj' | `disj' | `leqv'
\alt `limp' | `lnot' | `lxor' | `snot' | `wnot'

<conditional-operator> ::= `if' | `default' | `ite' | `hasValue'

<float-operator> ::=
~\alt <comp-operator> | `sqrt' | `cos' | `sin' | `tan'
\alt `acos' | `asin' | `atan' | `abs' | `minus'
\alt `add' | `sub' | `mult' | `fdiv' | `pow' | `floor'

<int-operator> ::=
~\alt <comp-operator> | `add' | `sub' | `mult'
\alt `div' | `pow' | `abs' | `minus' | `max' | `min'

<multimap-operator> ::=
~\alt <eq-operator> | `find' | `multimap_fold' | `isin' | `multimap_make'
\alt `countKeys' | `countEntries' |`sumIntEntries'
\alt `maxEntries' | `minEntries'

<set-operator> ::=
~\alt <eq-operator> | `union' | `inter' | `subset' | `makeSet'
\alt `isin' | `notin' | `length' | `set_fold'

<string-operator> ::= <eq-operator> | `concat' | `length'

<lambda-expr> ::= `lambda' `(' <term-list> `,' <expression> `)'

<tuple-expr> ::= `(' `)' | `(' <expressions> `)'

<expression> ::=
~\alt `val' `(' <type> `,' <term> `)'
\alt `variable' `(' <term> `)'
\alt `operation' `(' <operator> `,' <expression-list> `)'
\alt <lambda-expr>
\alt <tuple-expr>

<expression-list> ::= `(' `)' | `(' <expression> `,' <expression-list> `)'

<expressions> ::= <expression> | <expression> `,' <expressions>

\end{grammar}

\subsection{Building statement terms}

\begin{grammar}
<statement> ::=
~\alt `assert' `(' <expression> `)'
\alt `assign' `(' <term> `,' <expression> `)'
\alt `if' `(' <expression> `,' <statement> `,' <statement> `)'
\alt `noop'
\alt `statement_python' `(' <str> `)'
\alt `seq2' `(' <statement> `,' <statement> `)'
\alt `while' `(' <int> `,' <expression> `,' <statement> `)'

<domain> ::= `boolDomain' | `fromFacts' | `fromList' `(' <expression-list> `)'

<variable-atom> ::=
~\alt `variable_declare' `(' <term> `,' <term> `,' <domain> `)'
\alt `variable_declareOptional' `(' <term> `)'
\alt `variable_define' `(' <term> `,' <term> `,' <expression> `)'
\alt `variable_domain' `(' <term> `,' <expression> `)'

<multimap-atom> ::=
~\alt `multimap_declare' `(' <term> `,' <term> `)'
\alt `multimap_assign' `(' <term> `,' <term> `,' <expression> `,' <expression> `)'

<set-atom> ::=
~\alt `set_declare' `(' <term> `,' <term> `)'
\alt `set_assign' `(' <term> `,' <term> `,' <expression> `)'

<execution-atom> ::=
~\alt `execution_declare' `(' <term> `,' <term> `,' <statement> `,' <term-list> `,' <term-list> `)'
\alt `execution_run' `(' <term> `,' <term> `)'

<optimize-atom> ::=
~\alt `optimize_maximizeSum' `(' <term> `,' <expression> `,' <term> `)'
\alt `optimize_precision' `(' <expression>  `)'

<preference-atom> ::=
~\alt `preference_maximizeScore'
\alt `preference_holds' `(' <term> `,' <expression> `)'
\alt `preference_holds' `(' <term> `,' <expression> `,' <int> `)'
\alt `preference_variableValue' `(' <term> `,' <term> `,' <expression> `)'
\alt `preference_variableValue' `(' <term> `,' <term> `,' <expression> `,' <int> `)'

<atom> ::=
~\alt`assign' `(' <term> `,' <term> `,' <expression> `)'
\alt `ensure' `(' <term> `,' <expression> `)'
\alt `compute' `(' <term> `,' <val-list> `)'
\alt `computed' `(' <term> `,' <val-list> `,' <type> `,' <term> `)'
\alt <variable-atom>
\alt <multimap-atom>
\alt <set-atom>
\alt <execution-atom>
\alt <optimize-atom>
\alt <preference-atom>
\end{grammar}

\subsection{Result fact format}
\begin{grammar}

<variable-warning> ::=
~\alt `emptyDomain'
\alt `multipleDeclarations'
\alt `multipleDefinitions'
\alt `undeclared'

<warning-symbol> ::=
~\alt `variable' `(' <variable-warning> `)'

<atom> ::=
~\alt `value' `(' <term> `,' <type> `,' <term> `)'
\alt `set_value' `(' <term> `,' <type> `,' <term> `)'
\alt `multimap_value' `(' <term> `,' <type> `,' <term> `,' <type> `,' <term> `)'
\alt `preference_score' `(' <int> `)'
\alt `warning' `(' <term> `)'
\alt `warning' `(' <warning-symbol> `,' <term-list> `,' <term> `)'
\end{grammar}

\subsection*{Examples}
The following two assignments and two checks
\begin{align*}
\pi &\eqdef 3.14159\\
y &\eqdef (x+2) \times \pi \\
& y / 3 \geq 10
& y / 3 \leq 15
\end{align*}
\begin{verbatim}
assign(example1_pi,pi,val(float,float("3.14159"))).
assign(example1_y,y,operation(mult,
                             (operation(plus,
                                       (variable(x),
                                       (val(int,2),()))),
                             (variable(pi),())))).
ensure(example_check1,operation(geq,
                               (operation(div,
                                         (variable(y),
                                         (val(int,3),()))),
                               (val(int,10),())))).
ensure(example_check2,operation(leq,
                               (operation(div,
                                         (variable(y),
                                         (val(int,3),()))),
                               (val(int,15),())))).
\end{verbatim}

\paragraph{Other examples}

\begin{verbatim}
operation(python(val(str,"maths.cos")),(val(float,float("0.1")),()))
operation(python(val(str,"lambda x,y: x+y*y")),
         (variable(v1),
         (val(float,float("0.1")),())))

ensure(c1,operation(eq,
                   (variable(v1),
                   (variable(v2),
                   ()))))
\end{verbatim}

\paragraph{default function}

\begin{verbatim}
assign(assign_x0,x0,val(int,0)).
assign(assign_x1,x1,operation(if,(variable(cond),(val(int,1),())))).
assign(assign_y0,y0,operation(default,(variable(x0),(variable(x1),())))).
\end{verbatim}
