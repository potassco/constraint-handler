\subsection*{Syntax}
\begin{grammar}
%% <statement> ::= <ident> `=' <expr>
%% \alt `for' <ident> `=' <expr> `to' <expr> `do' <statement>
%% \alt `{' <stat-list> `}'
%% \alt <empty>

%% <stat-list> ::= <statement> `;' <stat-list> | <statement>

<type> ::= `bool' | `float' | `int' | `string' | `symbol' | `function' | `set'

<bool> ::= `true' | `false'

<int> ::= any integer literal

<str> ::= any string literal (enclosed in quotes)

<name> ::= any suitable string

<term> ::= <int> | <str> | <symbol>

<terms> ::= <term> | <term> `,' <terms>

<symbol> ::= <name> | <name> `(' <terms> `)'

<tterm-list> ::= `(' `)' | `(' <term> `,' <term-list> `)'

%<constant> ::= `bool' `(' <bool> `)' \alt `int' `(' <int> `)' \alt `float' `(' <str> `)' \alt `string `(' <str> `)' \alt `symbol `(' <term> `)'

<operator> ::= `python' `(' <expr> `)'
\alt <lambda-expr>
\alt <variable>
\alt <bool-operator>
\alt <conditional-operator>
\alt <float-operator>
\alt <int-operator>
\alt <multimap-operator>
\alt <set-operator>
\alt <string-operator>

<eq-operator> ::= `eq' | `neq'

<comp-operator> ::= <eq-operator> | `leq' | `lt' | `geq' | `gt'

<bool-operator> ::= <eq-operator> | `lnot' | `conj' | `disj' | `limp' | `lxor' | `leqv'

<conditional-operator> ::= `if' | `phi' | `ite'

<float-operator> ::= <comp-operator> | `sqrt' | `cos' | `sin' | `tan' | `acos' | `asin' | `atan' | `abs' | `minus'
| `add' | `sub' | `mult' | `div' | `pow' | `floor'

<int-operator> ::= <comp-operator> | `add' | `sub' | `mult' | `div' | `pow' | `abs' | `minus' | `max' | `min'

<multimap-operator> ::= TODO

<set-operator> ::= <eq-operator> | `union' | `inter' | `subset' | `makeSet' | `isin' | `notin' | `length' | `fold'

<string-operator> ::= <eq-operator> | `concat' | `length'

<lambda-expr> ::= `lambda' `(' <term-list> `,' <expr> `)'

<expr> ::=
`val' `(' <type> `,' <term> `)'
\alt `variable' `(' <term> `)'
\alt `operation' `(' <operator> `,' <expr-list> `)'
\alt <lambda-expr>

<expr-list> ::= `(' `)' | `(' <expr> `,' <expr-list> `)'

<atom> ::= `assign' `(' <term> `,' <term> `,' <expr> `)'
\alt `ensure' `(' <term> `,' <expr> `)'
\alt `set_declare' `(' <term> `,' <term> `)'
\alt `set_assign' `(' <term> `,' <term> `,' <expr> `)'
\alt `compute' `(' <term> `,' <val-list> `)'
\alt `computed' `(' <term> `,' <val-list> `,' <type> `,' <term> `)'
\end{grammar}

\subsection*{Examples}
The following two assignments and two checks
\begin{align*}
\pi &\eqdef 3.14159\\
y &\eqdef (x+2) \times \pi \\
& y / 3 \geq 10
& y / 3 \leq 15
\end{align*}
\begin{verbatim}
assign(example1_pi,pi,val(float,float("3.14159"))).
assign(example1_y,y,operation(mult,
                             (operation(plus,
                                       (variable(x),
                                       (val(int,2),()))),
                             (variable(pi),())))).
ensure(example_check1,operation(geq,
                               (operation(div,
                                         (variable(y),
                                         (val(int,3),()))),
                               (val(int,10),())))).
ensure(example_check2,operation(leq,
                               (operation(div,
                                         (variable(y),
                                         (val(int,3),()))),
                               (val(int,15),())))).
\end{verbatim}

\paragraph{Other examples}

\begin{verbatim}
operation(python(val(str,"maths.cos")),(val(float,float("0.1")),()))
operation(python(val(str,"lambda x,y: x+y*y")),
         (variable(v1),
         (val(float,float("0.1")),())))

ensure(c1,operation(eq,
                   (variable(v1),
                   (variable(v2),
                   ()))))
\end{verbatim}

\paragraph{phi function}

\begin{verbatim}
assign(assign_x0,x0,val(int,0)).
assign(assign_x1,x1,operation(if,(variable(cond),(val(int,1),())))).
assign(assign_y0,y0,operation(phi,(variable(x0),(variable(x1),())))).
\end{verbatim}
